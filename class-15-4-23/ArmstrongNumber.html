problem statement :

Write a program to check armstrong number.

Code : 

            function checkArmstrong(num){
                let result = 0;
                let tmpNumber = num;
                let power = String(num).length;
                while(tmpNumber > 0){
                    let rem = tmpNumber % 10;
                    result = result + rem**power;
                    tmpNumber = parseInt(tmpNumber / 10);
                }
                if(num == result){
                    console.log(`It is armstrong number.`);
                }
                else{
                    console.log(`It is not armstrong number.`);
                }
            }
            checkArmstrong(153);
