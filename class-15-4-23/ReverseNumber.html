problem statement :

Write a program to reverse a number.

Code : 

            function reverseNumber(num){
                let sum = 0;
                let tempNumber = num;
                while(tempNumber > 0){
                    let rem = tempNumber % 10;
                    sum = (sum * 10) + rem;
                    tempNumber = parseInt(tempNumber / 10);
                }
                console.log(`The Reverse of ${num} is ${sum}`);
            }
            reverseNumber(598);
