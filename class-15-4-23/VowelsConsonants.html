problem statement :

Write a program finding number of consonants and vowels.

Code : 

            function getCount(str){
                let string = str.replaceAll(" ","");
                const arr = ['a','e','i','o','u','A','E','I','O','U'];
                let vowel = 0;
                let consonant = 0;
                for(let i=0; i < string.length; i++){
                    if(arr.includes(string.charAt(i))){
                        vowel++;
                    }
                    else{
                        consonant++
                    }
                }
                console.log(`Number of vowels ${vowel} and consonants are ${consonant} present in the ${str}`);
            }
            getCount("Hi am Musharruf");
