problem statement :

Write a program to reverse a string.

Code : 

            function  reverseString(str){
                string = "";
                for(let i=str.length-1; i >= 0; i--){
                    string += str[i];
                }
                console.log(`The Reverse of ${str} is ${string}`);
            }
            reverseString("javascript");
