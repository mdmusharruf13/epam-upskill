Problem Statement :

Printing Array elements with Comma delimiters.

Code : 

            var arr = ['h','o','l','a'];
            
            function print(){
                var sum ="";
                for(var i=0; i < arr.length; i++){
                    if(i == arr.length-1){
                        sum += arr[i] ;
                    }
                    else{
                        sum += arr[i] +",";
                    }
                }
                return sum;
            }

            var ans = print();
            console.log("\""+ans+"\"");
