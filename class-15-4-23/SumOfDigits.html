problem statement :

Write a program for sum of individual digit a number.

Code : 

            function getSumOfDigits(num){
                let sum = 0;
                let tempNumber = num;
                while(tempNumber > 0){
                    let rem = tempNumber % 10;
                    sum = sum + rem;
                    tempNumber = parseInt(tempNumber / 10);
                }
                console.log(`The Sum of individual digit of ${num} is ${sum}`);
            }
            getSumOfDigits(598);
