problem statement :

Write a program for Even or Odd Number.

Code : 

            function checkEvenOdd(num){
                if(num%2 == 0){
                    console.log(`${num} is Even Number`);
                }
                else{
                    console.log(`${num} is Odd Number`);
                }
            }
            checkEvenOdd(598);
